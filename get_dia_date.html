METHOD get_dia_date BY DATABASE PROCEDURE FOR HDB LANGUAGE SQLSCRIPT
                    OPTIONS READ-ONLY USING  mara vbak vbpa BUT000 vbap VBKD vbep ZSD_KONV_TAB  MAKT.
  IT_VBAP = apply_filter ( vbap , :VBELN_STR );

             it_data =    SELECT
                               OdPlant,
                               OdTyp,
                               OdNo,
                               OdSr,
                               OdSku,
                               OdDmSz,
                               OdCrtDt,
                               OdqOpenQty,
                               BPSPCd,
                               BPSPNm,
                               BPENCd,
                               BPENNm,
                                BmKt,
                                BmCol,
                                DQLY,
                                BMStnWt,
                                BMStnQty,
                                BMDIAMWt,
                                BMDIAMQty,
                                BMLBGRWt,
                                BMLBGRQty,
                                BMCLSTWt,
                                BMCLSTQty,
                                BMCUBZWt,
                                BMCUBZQty,
                                 MQLT
                                ,BMATWt
                                ,BMATQty
                                ,BmDiaIns
                                ,BmDsgIns
                                ,BmCusIns
                                ,OdStIns
                                ,OdRhIns
                                ,OdSzIns

                            FROM (

                            SELECT S.MANDT AS OdClient,CAST(S.WERKS AS VARCHAR) AS OdPlant,S.ABGRU AS OdCanc
                                        ,SH.AUART AS OdTyp
                                        ,CAST(S.VBELN AS NUMERIC) AS OdNo,CAST(S.POSNR AS NUMERIC) AS OdSr,S.MATNR AS OdSku
                                        ,(CASE WHEN S.ZZSIZE1='' THEN (CASE WHEN SM.ZZSIZE1='' THEN SM.ZZPROD_LEN ELSE SM.ZZSIZE1 END) ELSE S.ZZSIZE1 END ) AS OdDmSz
                                        ,S.KWMENG AS OdOrdQty
                                        ,S.STAMP_INST AS OdStIns
                                        ,S.RHOD_INST AS OdRhIns
                                        ,S.SIZE_INST AS OdSzIns
                                        ,S.ERDAT AS OdCrtDt,Odqedatu_min,Odqedatu_max,OdqOpenQty,OdqBlkQty,OdqNRelQty,OdqTotQty
                                        ,IFNULL(BPLSPCd,BPHSPCd) AS BPSPCd,IFNULL(BPLSPNm,BPHSPNm) AS BPSPNm,IFNULL(BPLENCd,BPHENCd) AS BPENCd,IFNULL(BPLENNm,BPHENNm) AS BPENNm
                                        ,S.DIAM_INST AS BmDiaIns,
                                         S.DES_PRD_INST AS BmDsgIns,
                                         S.CUST_PRD_INST AS BmCusIns,
                                         (
                                            SELECT STRING_AGG(MSgrp1, ',') from
                                            (select MAX(CASE WHEN Q.MTART IN ('DIAM','LBGR','CLST','CUBZ') THEN DM.EXTWG ELSE '' END) AS MSGrp1 from ZSD_KONV_TAB AS Q
                                                                JOIN MARA DM  ON DM.MANDT=Q.MANDT AND DM.MATNR = Q.IDNRK
                                                                JOIN MAKT AS MA ON MA.MATNR = Q.MATNR
                                                 WHERE Q.MANDT = 100
                                                     AND Q.MANDT=OrClnt
                                                     AND Q.MATNR=OrSku
                                                     AND Q.ITM_CNT = BMITMCAT
                                                     AND Q.MTART IN ('DIAM','LBGR','CLST','CUBZ')
                                                GROUP BY DM.EXTWG
                                                ) AS YY
                                          ) AS DQLY
                                        ,BMStnWt,BMStnQty,DiaWT_PRPC AS BMDIAMWt,BMDIAMQty,BMLBGRWt,BMLBGRQty,BMCLSTWt,BMCLSTQty,BMCUBZWt,BMCUBZQty
                                        , (
                                            SELECT STRING_AGG(MSgrp1, ',') from
                                            (select MAX(CASE WHEN Q.MTART IN ('METL') THEN DM.EXTWG ELSE '' END) AS MSGrp1 from ZSD_KONV_TAB AS Q
                                                                JOIN MARA DM  ON DM.MANDT=Q.MANDT AND DM.MATNR = Q.IDNRK
                                                                JOIN MAKT AS MA ON MA.MATNR = Q.MATNR
                                                 WHERE Q.MANDT = 100
                                                     AND Q.MANDT=OrClnt1
                                                     AND Q.MATNR=OrSku1
                                                     AND Q.MTART IN ('METL')
                                                GROUP BY DM.EXTWG
                                                ) AS zz
                                          ) AS MQLT
                                        ,BMATWt,BMATQty
                                        ,BMKT
                                        ,BMCOL
                                        from :IT_VBAP S
                                        JOIN MARA SM ON SM.MANDT=S.MANDT AND SM.MATNR=S.MATNR
                                        JOIN VBAK SH ON SH.MANDT=S.MANDT AND SH.VBELN=S.VBELN
                                        LEFT JOIN VBKD PH ON PH.MANDT=S.MANDT AND PH.VBELN=S.VBELN AND PH.POSNR='000000'
                                        LEFT JOIN VBKD PL ON PL.MANDT=S.MANDT AND PL.VBELN=S.VBELN AND PL.POSNR=S.POSNR
                                        LEFT JOIN (
                                                        SELECT OBPH.MANDT AS BPHClient,OBPH.VBELN AS BPHOdNo,OBPH.POSNR AS BPHOdSr--,OBPH.KUNNR AS BPHCustCd,OBPH.PARVW AS BPHTyp
                                                              ,MAX(CASE WHEN OBPH.PARVW='AG' THEN OBPH.KUNNR ELSE '' END) AS BPHSPCd
                                                              ,MAX(CASE WHEN OBPH.PARVW='AG' THEN BPH.name_org1 ELSE '' END) AS BPHSPNm
                                                              ,MAX(CASE WHEN OBPH.PARVW='EN' THEN OBPH.KUNNR ELSE '' END) AS BPHENCd
                                                              ,MAX(CASE WHEN OBPH.PARVW='EN' THEN BPH.name_org1 ELSE '' END) AS BPHENNm
                                                        FROM VBPA OBPH
                                                        LEFT JOIN BUT000 BPH on BPH.PARTNER = OBPH.KUNNR
                                                        WHERE OBPH.MANDT=100 and OBPH.POSNR = '000000' AND OBPH.PARVW IN ('EN','AG')
                                                        GROUP BY OBPH.MANDT,OBPH.VBELN,OBPH.POSNR
                                                  ) AS OBPH on OBPH.BPHClient = S.MANDT AND OBPH.BPHOdNo = S.VBELN
                                        LEFT JOIN (
                                                        SELECT OBPL.MANDT AS BPLClient,OBPL.VBELN AS BPLOdNo,OBPL.POSNR AS BPLOdSr--,OBPL.KUNNR AS BPLCustCd,OBPL.PARVW AS BPLTyp
                                                              ,MAX(CASE WHEN OBPL.PARVW='AG' THEN OBPL.KUNNR ELSE '' END) AS BPLSPCd
                                                              ,MAX(CASE WHEN OBPL.PARVW='AG' THEN BPH.name_org1 ELSE '' END) AS BPLSPNm
                                                              ,MAX(CASE WHEN OBPL.PARVW='EN' THEN OBPL.KUNNR ELSE '' END) AS BPLENCd
                                                              ,MAX(CASE WHEN OBPL.PARVW='EN' THEN BPH.name_org1 ELSE '' END) AS BPLENNm
                                                        FROM VBPA OBPL
                                                        LEFT JOIN BUT000 BPH on BPH.PARTNER = OBPL.KUNNR
                                                        WHERE OBPL.MANDT=100 and OBPL.POSNR <> '000000' AND OBPL.PARVW IN ('EN','AG')
                                                        GROUP BY OBPL.MANDT,OBPL.VBELN,OBPL.POSNR
                                                  ) AS OBPL on OBPL.BPLClient = S.MANDT AND OBPL.BPLOdNo = S.VBELN AND OBPL.BPLOdSr = S.POSNR

                                      JOIN  ( SELECT SD.MANDT AS SDClient,SD.VBELN as OdqNo,SD.POSNR as OdqSr
                                                        ,min(case when lifsp = '' then edatu else '00000000' end) as Odqedatu_min
                                                        ,max(case when lifsp = '' then edatu else '00000000' end) as Odqedatu_max
                                                        ,max(vrkme) as Odqvrkme ,sum(wmeng) as Odqwmeng,sum(cmeng) as Odqcmeng,sum(bmeng) as Odqbmeng
                                                        ,sum(wmeng)-(sum(case when lifsp <> '' then wmeng else 0 end))    as OdqOpenQty
                                                        ,sum(case when lifsp <> '' then wmeng else 0 end) as OdqBlkQty
                                                        ,sum(case when lifsp = '' and ettyp = 'CN' then wmeng else 0 end) as OdqNRelQty
                                                        ,sum(wmeng) as OdqTotQty
                                                FROM  VBEP SD
                                                GROUP BY SD.MANDT,SD.VBELN,SD.POSNR --,SD.REQ_DLVDATE
                                        ) AS SD ON SD.SDClient=S.MANDT
                                               AND SD.OdqNo=S.VBELN
                                               AND SD.OdqSr=S.POSNR

                                LEFT JOIN ( SELECT                O.MANDT AS OrClnt,
                                                                  O.VBELN AS ORSO,
                                                                  O.POSNR AS ORSR,
                                                                  O.MATNR AS OrSku,
                                                                  O.ITM_CNT AS BMITMCAT,
                                                              ---    M.MTART as mat_typ,
                                                                 --- O.IDNRK AS OrRmCd,
                                                                  SUM(CASE WHEN M.MTART IN ('DIAM','LBGR','CLST','CUBZ') THEN O.MENGE  ELSE '0' END) AS BMStnWt,
                                                                  SUM(CASE WHEN M.MTART IN ('DIAM','LBGR','CLST','CUBZ') THEN O.NO_OF_STONE ELSE '0' END) AS BMStnQty,
                                                                 --   ,O.NO_OF_STONE * to_decimal((case when O.PER_STN_WT = '' then '0' else O.PER_STN_WT end) ,6,4 ) as DiaWT_PRPC
                                                                    SUM(O.MENGE) as DiaWT_PRPC
                                                                    ,SUM(CASE WHEN M.MTART='DIAM' THEN O.MENGE  ELSE '0' END) AS BMDIAMWt
                                                                    ,SUM(CASE WHEN M.MTART='DIAM' THEN O.NO_OF_STONE ELSE '0' END) AS BMDIAMQty
                                                                    ,SUM(CASE WHEN M.MTART='LBGR' THEN O.MENGE  ELSE '0' END) AS BMLBGRWt
                                                                    ,SUM(CASE WHEN M.MTART='LBGR' THEN O.NO_OF_STONE ELSE '0' END) AS BMLBGRQty
                                                                    ,SUM(CASE WHEN M.MTART='CLST' THEN O.MENGE  ELSE '0' END) AS BMCLSTWt
                                                                    ,SUM(CASE WHEN M.MTART='CLST' THEN O.NO_OF_STONE ELSE '0' END) AS BMCLSTQty
                                                                    ,SUM(CASE WHEN M.MTART='CUBZ' THEN O.MENGE  ELSE '0' END) AS BMCUBZWt
                                                                    ,SUM(CASE WHEN M.MTART='CUBZ' THEN O.NO_OF_STONE ELSE '0' END) AS BMCUBZQty
                                                                    ,O.BOM_KT AS BMKT
                                                                    ,O.BOM_COL AS BMCOL
                                                                    --,M.ZZEMR AS OLDEMRCD
                                                                FROM ZSD_KONV_TAB AS O
                                                                JOIN MARA M  ON M.MANDT=O.MANDT AND M.MATNR = O.IDNRK
                                                                JOIN MAKT AS MA ON MA.MATNR = O.MATNR
                                                                WHERE O.MANDT = 100 AND O.BOM_ITM_TYP IN ('RM') and O.DEL_FLAG = '' and M.MTART IN ('CLST','CUBZ','DIAM','LBGR')
                                                                GROUP BY  O.MANDT,O.VBELN ,O.POSNR,O.MATNR,O.BOM_KT,O.BOM_COL,O.ITM_CNT

                                     ) AS MS ON  OrClnt=S.MANDT AND OrSku=S.MATNR and ORSO=S.vbeln and ORSR=S.posnr

                                     LEFT JOIN (
                                                         SELECT  O.MANDT AS OrClnt1,
                                                                  O.VBELN AS ORSO1,
                                                                  O.POSNR AS ORSR1,
                                                                  O.MATNR AS OrSku1,
                                                                ---  M.MTART as mat_typ1,
                                                                 --- O.IDNRK AS OrRmCd,
                                                                  SUM(CASE WHEN M.MTART IN ('METL') THEN O.MENGE  ELSE '0' END) AS BMATWt,
                                                                  SUM(CASE WHEN M.MTART IN ('METL') THEN O.NO_OF_STONE ELSE '0' END) AS BMATQty,
                                                                 --   ,O.NO_OF_STONE * to_decimal((case when O.PER_STN_WT = '' then '0' else O.PER_STN_WT end) ,6,4 ) as DiaWT_PRPC
                                                                    SUM(O.MENGE) as DiaWT_PRPC1
                                                                    ,SUM(CASE WHEN M.MTART='METL' THEN O.MENGE  ELSE '0' END) AS BMDIAMWt1
                                                                    ,O.BOM_KT AS BMKT1
                                                                    ,O.BOM_COL AS BMCOL1
                                                                    --,M.ZZEMR AS OLDEMRCD
                                                                FROM ZSD_KONV_TAB AS O
                                                                JOIN MARA M  ON M.MANDT=O.MANDT AND M.MATNR = O.IDNRK
                                                                JOIN MAKT AS MA ON MA.MATNR = O.MATNR
                                                                WHERE O.MANDT = 100 AND O.BOM_ITM_TYP IN ('RM') and O.DEL_FLAG = '' and M.MTART IN ('METL')
                                                                GROUP BY  O.MANDT,O.VBELN ,O.POSNR ,O.MATNR ,O.BOM_KT,O.BOM_COL

                                     ) MS1 ON  OrClnt1=S.MANDT AND OrSku1=S.MATNR and ORSO1=S.vbeln and ORSR1=S.posnr

                                        WHERE S.MANDT=100 AND LEFT(S.WERKS,1)IN ('1','2') AND S.ABGRU='' --and S.VBELN = '0001007402' ---AND CAST(S.VBELN AS NUMERIC)='2000916'
                                        ORDER BY S.WERKS,S.VBELN,S.POSNR,S.MATNR
                                      ) AS M; --where OdSku = 'RE26668-PWDI1-HSF1';


endmethod.
